```python
import requests, os

r = requests.post("https://ninjachat.ai/api/v1/chat",
    headers={"Authorization": f"Bearer {os.environ['NINJACHAT_API_KEY']}"},
    json={"model": "grok-4", "messages": [{"role": "user", "content": "Hello!"}]}

print(r.json()["message"]["content"])
```
